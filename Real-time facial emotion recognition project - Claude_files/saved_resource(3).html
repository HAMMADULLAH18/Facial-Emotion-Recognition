Phase 1 — Webcam Capture & Face Detection

Start by getting frames from the webcam and detecting face bounding boxes. OpenCV's Haar Cascade is the quickest entry point; upgrade to MTCNN for accuracy.

utils/face_detector.py 
import cv2
import numpy as np

# Load OpenCV Haar Cascade (built-in, no download needed)
face_cascade = cv2.CascadeClassifier(
    cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
)

def detect_faces(frame):
    """Returns list of (x, y, w, h) bounding boxes."""
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    faces = face_cascade.detectMultiScale(
        gray,
        scaleFactor=1.1,
        minNeighbors=5,
        minSize=(48, 48)
    )
    return faces if len(faces) > 0 else []

def crop_face(frame, x, y, w, h, target=(48, 48)):
    """Crop + resize + grayscale a face ROI for model input."""
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    roi  = gray[y:y+h, x:x+w]
    roi  = cv2.resize(roi, target)
    roi  = roi.astype('float32') / 255.0
    roi  = np.expand_dims(roi, axis=(0, -1)) # (1, 48, 48, 1)
    return roi
Tip: For multi-face detection or profile faces, switch to from mtcnn import MTCNN; detector = MTCNN(). It handles rotated/angled faces far better than Haar Cascade.

Phase 2 — Emotion Classification Model

Two main approaches: use a pre-built library (fastest to start), or load a custom trained .h5 model on the FER-2013 dataset (more control).

Option A — Using the fer library (Recommended for Beginners)

model_loader.py — approach A 
from fer import FER
import cv2

detector = FER(mtcnn=True)  # uses MTCNN under the hood

def predict(frame):
    result = detector.detect_emotions(frame)
    if not result:
        return None
    emotions = result[0]['emotions']      # dict of 7 scores
    top_emotion = max(emotions, key=emotions.get)
    confidence  = emotions[top_emotion]
    return top_emotion, confidence, emotions, result[0]['box']

Option B — Custom Keras CNN on FER-2013

model_loader.py — approach B 
from tensorflow.keras.models import load_model
import numpy as np

LABELS = ['angry','disgust','fear','happy','neutral','sad','surprise']
model  = load_model('models/emotion_model.h5')

def predict_from_roi(roi):
    """roi shape: (1, 48, 48, 1), normalized 0-1"""
    preds = model.predict(roi, verbose=0)[0]   # shape (7,)
    idx   = np.argmax(preds)
    return LABELS[idx], float(preds[idx]), dict(zip(LABELS, preds))
Where to get a pretrained model: Search for "FER2013 pretrained h5" on Kaggle or HuggingFace. A well-trained VGG-style CNN on FER-2013 achieves ~65–72% test accuracy — this is near the human-level ceiling on that dataset.

Phase 3 — Live Overlay & Display

Draw bounding boxes, emotion labels, confidence bars, and optional mini emotion histograms directly onto OpenCV frames.

utils/overlay.py 
import cv2

COLORS = {
    'happy':    (50,  205, 50),
    'sad':      (220, 100, 30),
    'angry':    (30,  30,  220),
    'neutral':  (180, 180, 180),
    'surprise': (30,  200, 220),
    'fear':     (180, 60,  220),
    'disgust':  (60,  180, 30),
}

def draw_face(frame, x, y, w, h, emotion, conf):
    color = COLORS.get(emotion, (255,255,255))
    # Bounding box
    cv2.rectangle(frame, (x,y), (x+w,y+h), color, 2)
    # Label background
    label = f"{emotion.upper()}  {conf*100:.1f}%"
    (tw, th), _ = cv2.getTextSize(label, cv2.FONT_HERSHEY_SIMPLEX, 0.6, 2)
    cv2.rectangle(frame, (x, y-th-10), (x+tw+8, y), color, -1)
    cv2.putText(frame, label, (x+4, y-6),
                cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0,0,0), 2)
    # Confidence bar
    bar_w = int(w * conf)
    cv2.rectangle(frame, (x, y+h+4), (x+bar_w, y+h+12), color, -1)
    cv2.rectangle(frame, (x, y+h+4), (x+w,     y+h+12), color,  1)

def draw_fps(frame, fps):
    cv2.putText(frame, f"FPS: {fps:.1f}", (10,25),
                cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0,255,0), 2)

Main Loop

main.py 
import cv2, time
from utils.face_detector import detect_faces, crop_face
from model_loader import predict_from_roi
from utils.overlay import draw_face, draw_fps
from utils.logger import EmotionLogger

cap    = cv2.VideoCapture(0)
logger = EmotionLogger()
prev   = time.time()

while True:
    ret, frame = cap.read()
    if not ret: break

    faces = detect_faces(frame)
    for (x, y, w, h) in faces:
        roi = crop_face(frame, x, y, w, h)
        emotion, conf, all_scores = predict_from_roi(roi)
        draw_face(frame, x, y, w, h, emotion, conf)
        logger.log(emotion, conf, all_scores)

    now = time.time(); fps = 1 / (now - prev); prev = now
    draw_fps(frame, fps)

    cv2.imshow('FER — Press Q to quit', frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
logger.save()

Phase 4 — Stats Logging & Analytics

utils/logger.py 
import csv, json
from datetime import datetime
from collections import defaultdict

class EmotionLogger:
    def __init__(self):
        ts   = datetime.now().strftime("%Y%m%d_%H%M%S")
        self.csv_path  = f"logs/emotion_log_{ts}.csv"
        self.counts    = defaultdict(int)
        self.rows      = []

    def log(self, emotion, conf, all_scores):
        self.counts[emotion] += 1
        self.rows.append({
            'timestamp': datetime.now().isoformat(),
            'emotion': emotion,
            'confidence': round(conf, 4),
            **{k: round(float(v),4) for k,v in all_scores.items()}
        })

    def save(self):
        if not self.rows: return
        with open(self.csv_path, 'w', newline='') as f:
            writer = csv.DictWriter(f, fieldnames=self.rows[0].keys())
            writer.writeheader(); writer.writerows(self.rows)
        total = sum(self.counts.values())
        summary = {k: {'count': v, 'pct': round(v/total*100,1)}
                   for k,v in self.counts.items()}
        with open('logs/session_summary.json', 'w') as f:
            json.dump(summary, f, indent=2)
        print(f"[Logger] Saved {total} frames → {self.csv_path}")

import pandas as pd
import matplotlib.pyplot as plt, json, glob

df = pd.read_csv(sorted(glob.glob('logs/*.csv'))[-1])
df['timestamp'] = pd.to_datetime(df['timestamp'])

Model Options & Comparison

Approach Speed Accuracy Ease Notes
fer library ⭐ Recommended ~20 FPS CPU ~65% ⭐⭐⭐⭐⭐ Plug-and-play, great for getting started fast
DeepFace ~10 FPS CPU ~70% ⭐⭐⭐⭐ Wraps multiple backends (VGG-Face, OpenFace)
Custom FER-2013 CNN ~25 FPS GPU 65–72% ⭐⭐⭐ Full control, best for learning the internals

Recommended path: Start with the fer library to get a working demo quickly, then swap in a custom Keras model once you understand the pipeline. This way you always have a running project to show.

Roadmap & Enhancements

Once the core pipeline is running, here's how to level it up progressively:

Phase 1 — Core
Basic Working Demo
Webcam + face detection + single emotion label on screen. Get this working first, commit it.
Phase 2 — Polish
Confidence + Multi-face + Logging
Show all 7 emotion scores as a mini bar chart on overlay. Support multiple faces in frame. Add CSV logger.
Phase 3 — Analytics
Session Stats Dashboard
Run analyze_logs.py after each session. Generate pie charts, confidence trends, dominant emotion stats.
Phase 4 — Advanced
Temporal Smoothing
Average predictions over a rolling window of 5–10 frames to eliminate flickering. Use collections.deque.
Phase 5 — Bonus
Web UI / Alert System
Stream frames via Flask + Socket.IO to a browser dashboard. Add audio alert on detecting "angry" or "sad" emotion sustained for 5+ seconds.
